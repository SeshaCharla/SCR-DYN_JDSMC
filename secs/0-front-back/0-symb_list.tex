\section*{Nomenclature}

\begin{nomenclature}
    \entry{$b_{urea}$}{Urea-to-ammonia conversion factor}
    \entry{$\con{\bullet}$}{Concentration of species $\bullet$ $(mol/m^3)$}
    \entry{$\con{NH_3}^{ads}$}{Surface concentration of adsorbed ammonia $(mol/m^2)$}
    \entry{$\con{NH_3}^{in}$}{Inlet ammonia concentration $(mol/m^3)$}
    \entry{$\con{NH_3}^{out}$}{Outlet ammonia concentration $(mol/m^3)$}
    \entry{$\con{NO_x}^{in}$}{Inlet $NO_x$ concentration $(mol/m^3)$}
    \entry{$\con{NO_x}^{out}$}{Outlet $NO_x$ concentration $(mol/m^3)$}
    \entry{$f_s$}{Sampling frequency $(Hz)$}
    \entry{$F_{vol}$}{Volumetric flow rate of exhaust gases $(m^3/s)$}
    \entry{$g_{sat}(\sigma)$}{Saturation function for adsorbed ammonia}
    \entry{$\Gamma$}{Total surface concentration of adsorption sites $(mol/m^2)$}
    \entry{$k_{ads}$}{Ammonia adsorption rate constant}
    \entry{$k_{des}$}{Ammonia desorption rate constant}
    \entry{$k_{ox}$}{Ammonia oxidation (AMOX) rate constant}
    \entry{$k_{scr}$}{Standard SCR reaction rate constant}
    \entry{$\lr{\bullet}$}{Parentheses notation}
    \entry{$\lrb{\bullet}$}{Square brackets notation}
    \entry{$\lrf{\bullet}$}{Curly braces notation}
    \entry{$\mol{\bullet}$}{Number of moles of species $\bullet$ $(mol)$}
    \entry{$NH_3$}{Ammonia}
    \entry{$NO_x$}{Nitrogen oxides (NO and $NO_2$)}
    \entry{$T$}{Temperature $(K)$}
    \entry{$t_s$}{Sampling time $(s)$}
    \entry{$\tau$}{Residence time of exhaust flow through SCR-ASC system $(s)$}
    \entry{$\tau_{urea}$}{Time constant for urea-to-ammonia conversion $(s)$}
    \entry{$\theta_{free}$}{Free adsorption sites}
    \entry{$u_{inj}$}{Urea injection rate (control input)}
    \entry{$V_{scr}$}{Volume of the SCR catalyst chamber $(m^3)$}
\end{nomenclature}
